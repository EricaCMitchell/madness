148/// evaluates Wigner-Seitz-truncated potential in the simulation cell, due to periodic or nonperiodic source functions
149class WignerSeitzPotentialFunctor : public FunctionFunctorInterface<double,3> {
150public:
151 /// Constructs a WignerSeitzPotentialFunctor evaluating potential
152 /// in simulation cell \p c due to point charges \p atoms optionally
153 /// periodically repeated according to boundary conditions \p b
154 /// and interaction range along each Cartesian direction limited by \p r .
155 /// Lattice summation range along each axis can be overridden by specifying
156 /// \p lattice_sum_range .
157 /// \tparam Int
158 /// \param atoms list of point charges in the simulation cell
159 /// \param c the simulation cell dimensions
160 /// \param b the boundary conditions
161 /// \param r the kernel range
162 /// \param lattice_sum_range overrides the lattice summation range that by default
163 template <typename Int>
164 WignerSeitzPotentialFunctor(const Molecule &atoms, Tensor<double> c,
165 BoundaryConditions<3> b, std::array<KernelRange, 3> r,
166 std::array<Int, 3> lattice_sum_range)
167 : atoms(atoms), cell(std::move(c)), bc(std::move(b)), range(std::move(r)),
168 cell_width{cell(0, 1) - cell(0, 0), cell(1, 1) - cell(1, 0),
169 cell(2, 1) - cell(2, 0)},
170 rcell_width{1. / cell_width[0], 1. / cell_width[1],
171 1. / cell_width[2]}, lattice_sum_range{lattice_sum_range[0], lattice_sum_range[1], lattice_sum_range[2]} {
172 for (int d = 0; d != 3; ++d)
173 MADNESS_ASSERT(lattice_sum_range[d] >= 0);
174 }
175
176 /// same as the standard ctor, but lacks the ability to override the lattice sum range
177 WignerSeitzPotentialFunctor(const Molecule &atoms, Tensor<double> c,
178 BoundaryConditions<3> b, std::array<KernelRange, 3> r) :
179 // N.B. move is a cast, calling make_default_lattice_sum_range like this is OK
180 WignerSeitzPotentialFunctor(atoms, std::move(c), std::move(b), std::move(r), make_default_lattice_sum_range(b,r)) {}
181
182 double operator()(const coord_3d &x) const final;
183
184 std::vector<coord_3d> special_points() const final;
185
186 static std::array<std::int64_t, 3> make_default_lattice_sum_range(const BoundaryConditions<3>& bc, const std::array<KernelRange, 3>& range) {
187 std::array<std::int64_t, 3> result;
188 for (int d = 0; d != 3; ++d) {
189 result[d] = bc.is_periodic()[d] ? (range[d].iextent_x2() + 1) / 2 : 0;
190 }
191 return result;
192 }
193
194private:
195 const Molecule &atoms;
196 const Tensor<double> cell;
197 const BoundaryConditions<3> bc;
198 const std::array<KernelRange, 3> range;
199 const std::array<double, 3> cell_width;
200 const std::array<double, 3> rcell_width;
201 const std::array<std::int64_t, 3> lattice_sum_range; // range of lattice summation, default is # of cells in each direction with nonzero contributions to the simulation cell
202};

254 vector_real_function_3d core_projection(World & world, const vector_real_function_3d& psi, const bool include_Bc = true)
255 {
256 int npsi = psi.size();
257 if (npsi == 0) return psi;
258 int natom = mol.natom();
259 vector_real_function_3d proj = zero_functions_compressed<double,3>(world, npsi);
260 real_tensor overlap_sum(static_cast<long>(npsi));
261
262 for (int i=0; i<natom; ++i) {
263 Atom at = mol.get_atom(i);
264 unsigned int atn = at.atomic_number;
265 unsigned int nshell = mol.n_core_orb(atn);
266 if (nshell == 0) continue;
267 for (unsigned int c=0; c<nshell; ++c) {
268 unsigned int l = mol.get_core_l(atn, c);
269 int max_m = (l+1)*(l+2)/2;
270 nshell -= max_m - 1;
271 for (int m=0; m<max_m; ++m) {
272 real_function_3d core = real_factory_3d(world).functor(real_functor_3d(new CoreOrbitalFunctor(mol, i, c, m)));
273 real_tensor overlap = inner(world, core, psi);
274 overlap_sum += overlap;
275 for (int j=0; j<npsi; ++j) {
276 if (include_Bc) overlap[j] *= mol.get_core_bc(atn, c);
277 proj[j] += core.scale(overlap[j]);
278 }
279 }
280 }
281 world.gop.fence();
282 }
283 if (world.rank() == 0) print("sum_k <core_k|psi_i>:", overlap_sum);
284 return proj;
285 }

287 double core_projector_derivative(World & world, const vector_real_function_3d& mo, const real_tensor& occ, int atom, int axis)
288 {
289 vector_real_function_3d cores, dcores;
290 std::vector<double> bc;
291 unsigned int atn = mol.get_atom(atom).atomic_number;
292 unsigned int ncore = mol.n_core_orb(atn);
293
294 // projecting core & d/dx core
295 for (unsigned int c=0; c<ncore; ++c) {
296 unsigned int l = mol.get_core_l(atn, c);
297 int max_m = (l+1)*(l+2)/2;
298 for (int m=0; m<max_m; ++m) {
299 real_functor_3d func = real_functor_3d(new CoreOrbitalFunctor(mol, atom, c, m));
300 cores.push_back(real_function_3d(real_factory_3d(world).functor(func).truncate_on_project()));
301 func = real_functor_3d(new CoreOrbitalDerivativeFunctor(mol, atom, axis, c, m));
302 dcores.push_back(real_function_3d(real_factory_3d(world).functor(func).truncate_on_project()));
303 bc.push_back(mol.get_core_bc(atn, c));
304 }
305 }
306
307 // calc \sum_i occ_i <psi_i|(\sum_c Bc d/dx |core><core|)|psi_i>
308 double r = 0.0;
309 for (unsigned int c=0; c<cores.size(); ++c) {
310 double rcore= 0.0;
311 real_tensor rcores = inner(world, cores[c], mo);
312 real_tensor rdcores = inner(world, dcores[c], mo);
313 for (unsigned int i=0; i<mo.size(); ++i) {
314 rcore += rdcores[i] * rcores[i] * occ[i];
315 }
316 r += 2.0 * bc[c] * rcore;
317 }
318
319 return r;
320 }

330 void make_nuclear_potential(World& world) {
331 double safety = 0.1;
332 double vtol = FunctionDefaults<3>::get_thresh() * safety;
333 vnuc = real_factory_3d(world).functor(real_functor_3d(new MolecularPotentialFunctor(mol))).thresh(vtol).truncate_on_project();
334 vnuc.set_thresh(FunctionDefaults<3>::get_thresh());
335 vnuc.reconstruct();
336 // "" is legacy core_type value for all-electron (also be used by CorePotentialManager)
337 // "none" is current core_type value for all-electron
338 if (core_type_ != "" && core_type_ != "none") {
339 real_function_3d c_pot = real_factory_3d(world).functor(real_functor_3d(new MolecularCorePotentialFunctor(mol))).thresh(vtol).initial_level(4);
340 c_pot.set_thresh(FunctionDefaults<3>::get_thresh());
341 c_pot.reconstruct();
342 vnuc += c_pot;
343 vnuc.truncate();
344 }
345 }
